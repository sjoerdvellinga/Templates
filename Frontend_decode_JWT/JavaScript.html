// Add following function to javascript of index class="html:" function parseJwt (token) { // https://stackoverflow.com/questions/38552003/how-to-decode-jwt-token-in-javascript var base64Url = token.split('.')[1]; var base64 = decodeURIComponent(atob(base64Url).split('').map((c)=>{ return '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2); }).join('')); return JSON.parse(base64); };
